ship_event = {
	id = ancrel.1
	title = "ancrel.1.name"
	desc = "ancrel.1.desc"
	picture = GFX_evt_overgrown_city
	location = from
	show_sound = event_wind_ruins

	fire_only_once = yes
	
	is_triggered_only = yes
	
	trigger = {
		host_has_dlc = "Ancient Relics Story Pack"
		owner = { is_ai = no }
		solar_system = { has_star_flag = precursor_zroni_1 }
		solar_system = { NOT = { is_same_value = root.owner.capital_scope.solar_system } }
		NOT = { has_global_flag = zroni_started }
		FROM = {
			has_anomaly = no
			NOT = { exists = archaeological_site }
			NOT = { has_planet_flag = suppress_archaeological_sites }
###expanded events mod
			OR = {
				habitable_planet = yes
				is_nuked = yes
				has_modifier = terraforming_candidate
			}
		}
###
		owner = {
			NOR = {
				has_country_flag = baol_intro
				has_country_flag = vultaum_intro
				has_country_flag = yuht_intro
				has_country_flag = first_league_intro
				has_country_flag = irassian_intro
				has_country_flag = cybrex_intro
				has_country_flag = zroni_intro
			}
		}
	}
	
	immediate = {
		owner = { set_country_flag = zroni_intro }
		set_global_flag = zroni_started
		solar_system = { set_star_flag = first_zoroni_site@root.owner }
		from = { prevent_anomaly = yes }
	}
	
	option = {
		name = CURIOUS
		hidden_effect = {
			owner = { country_event = { id = story.5 days = 30 } }
		}
		from = { 
			create_archaeological_site = zroni_digsite_1
			root.owner = {
				set_country_flag = zroni_arc_chain_@prev
			}
			add_deposit = d_zro_deposit_2
		}
		owner = {
			add_monthly_resource_mult = {
				resource = engineering_research
     			value = @tier1researchreward
     			min = @tier1researchmin
     			max = @tier1researchmax
     		}
		}
	}
}

fleet_event = { 
	id = ancrel.4
	title = "ancrel.3.name"
	desc = ancrel.4.desc
	picture = GFX_evt_zro_1
	show_sound = event_energy_field
	location = from

	archaeology = yes

	is_triggered_only = yes
	
	immediate = {
		closest_system = {
			limit = {
				has_access_fleet = root.owner
				NOR = {
					is_same_value = root.solar_system
					has_star_flag = first_zoroni_site@root.owner
					is_same_value = root.owner.capital_scope.solar_system
					has_star_flag = hostile_system
				}
				is_fe_cluster = no
				OR = {
					has_owner = no
					owner = { is_same_empire = root.owner }
				}
				any_system_planet = {
					habitable_planet = yes
					is_colony = no
					has_anomaly = no
					NOR = { 
						exists = archaeological_site
						has_planet_flag = hatchling_egg
						has_planet_flag = suppress_archaeological_sites #for modders
					}
				}
				root.owner = {
					any_owned_fleet = {
						is_ship_class = shipclass_science_ship
						can_access_system = prevprev
					}
				}
			}
			min_steps = 1
			max_steps = 15
			random_system_planet = {
				limit = {
					habitable_planet = yes
					has_anomaly = no
					is_colony = no
					NOR = { 
						exists = archaeological_site
						has_planet_flag = hatchling_egg
						has_planet_flag = suppress_archaeological_sites
					}
				}
				create_archaeological_site = zroni_digsite_2
				save_event_target_as = zroni_expanded_system ####expanded events mod

				root.owner = {
					set_country_flag = zroni_arc_chain_@prev
				}
				add_deposit = d_zro_deposit_3
				save_event_target_as = new_zroni_digsite
			}
		}
		if = {
			limit = {
				NOT = { exists = event_target:new_zroni_digsite }
			}
			closest_system = {
				limit = {
					has_access_fleet = root.owner
					NOR = {
						is_same_value = root.solar_system
						has_star_flag = first_zoroni_site@root.owner
						is_same_value = root.owner.capital_scope.solar_system
						has_star_flag = hostile_system
					}
					is_fe_cluster = no
					OR = {
						has_owner = no
						owner = { is_same_empire = root.owner }
					}
					any_system_planet = {
						OR = {
							is_planet_class = pc_barren
							is_planet_class = pc_barren_cold
						}
						has_anomaly = no
						is_colony = no
						NOR = { 
							exists = archaeological_site
							has_planet_flag = hatchling_egg
							has_planet_flag = suppress_archaeological_sites
						}
					}
				}
				min_steps = 1
				random_system_planet = {
					limit = {
						OR = {
							is_planet_class = pc_barren
							is_planet_class = pc_barren_cold
						}
						has_anomaly = no
						is_colony = no
						NOR = { 
							exists = archaeological_site
							has_planet_flag = hatchling_egg
							has_planet_flag = suppress_archaeological_sites
						}
					}
					if = { ###expanded events mod

						limit = { can_be_terraforming_candidate = yes }
						add_modifier = { modifier = "terraforming_candidate" days = -1 }	
					}
					create_archaeological_site = zroni_digsite_2
					save_event_target_as = zroni_expanded_system ###expanded events mod

					root.owner = {
						set_country_flag = zroni_arc_chain_@prev
					}
					add_deposit = d_zro_deposit_3
					save_event_target_as = new_zroni_digsite
				}
			}
		}
	}

	option = {
		name = OK
		small_artifact_reward = yes
		owner = { ###expanded events mod

			add_resource = { minor_artifacts = 5 }
			create_point_of_interest = {
					id = zroni_system_poi.1
					name = "zroni_system_1_poi"
					desc = "zroni_system_1_poi_desc"
					event_chain = "zroni_chain"
					location = event_target:zroni_expanded_system
				}
		}
		owner = {
			IF = {
				limit = {
					is_regular_empire = yes
					NOR = { 
						has_technology = tech_psionic_theory
						has_tech_option = tech_psionic_theory
					}
				}
				add_research_option = tech_psionic_theory
				add_tech_progress = {
					tech = tech_psionic_theory
					progress = 0.25
				}
			}
			ELSE_IF = { ###expanded events mod

				limit = {
					has_tech_option = tech_psionic_theory
				}
				add_tech_progress = {
					tech = tech_psionic_theory
					progress = 0.5
				}
			}
			ELSE = {
				add_monthly_resource_mult = {
					resource = society_research
					value = @tier4researchreward
					min = @tier4researchmin
					max = @tier4researchmax
				}
			}
		}
	}
