 # Ruler gets Vision trait back when no longer ruler
country_event = {
    id = akillian.200
    is_triggered_only = yes
    hide_window = yes

    trigger = {
        from = {
            has_leader_flag = vision_leader
        }
    }

    immediate = {
        from = {
            if = {
                limit = { leader_class = scientist }
                add_trait = leader_trait_vision
                break = yes
            }
        }
    }
}
