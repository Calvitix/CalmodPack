				set_country_flag = lcluster_clue_pirate

							leader = {

									has_trait = leader_trait_expertise_computing

### Junkyard
ship_event = {
	id = anomaly.30
	title = "anomaly.30.name"
#Exp. events
	desc = {
		trigger = {
			NOT = { has_ship_flag = anomaly_30_clue }
		}
		text = "anomaly.30.desc"
	}
	desc = {
		trigger = {
			has_ship_flag = anomaly_30_clue
		}
		text = "anomaly.30.desc.clue"
	}
###
	picture = GFX_evt_junkheap
	show_sound = event_ship_bridge
	location = FROM

	is_triggered_only = yes

	immediate = {
		from = { clear_deposits = yes }
		owner = { country_event = { id = story.5 days = 30 } }
#Exp. events
		if = {
			limit = {
				has_distar = yes
				NOT = { has_global_flag = l_cluster_opened }
				owner = { has_event_chain = l_cluster_chain }
				leader = {
					OR = {
						has_trait = leader_trait_meticulous
						has_trait = leader_trait_cataloger
						has_trait = leader_trait_expertise_materials
					}
				}
			}
			set_ship_flag = anomaly_30_clue
		}
###
	}

	option = {
		name = anomaly.30.a
#Exp. events
		owner = {
			hidden_effect = {
				set_country_flag = lcluster_clue_junkyard
				pop_lcluster_event = yes
			}
		}
	}
	option = {
		name = FASCINATING
		trigger = {
			has_ship_flag = anomaly_30_clue
		}
		allow = {
			leader = {
				OR = {
					has_trait = leader_trait_meticulous
					has_trait = leader_trait_cataloger
					has_trait = leader_trait_expertise_materials
				}
			}
		}
		owner = {	
			generate_lcluster_clue = yes
			add_resource = { nanites = 250 }
		}
	}
	after = {
		FROM = { 
			set_deposit = d_minerals_3
		}
		if = {
			limit = { has_ship_flag = anomaly_30_clue }
			FROM = { 
				add_deposit = d_nanites_deposit
			}
		}
		else = {
			FROM = { 
				add_deposit = d_alloys_1
			}
		}
		hidden_effect = {
			FROM = {
				add_modifier = {
					modifier = "junkyard"
					days = -1
				}
			}
			remove_ship_flag = anomaly_30_clue
		}
		owner = { small_artifact_chance = yes }
	}
}

### Living Metal
ship_event = {
	id = anomaly.2576
	title = "anomaly.2576.name"
	desc = "anomaly.2576.desc"
	picture = GFX_evt_mining_operations
	location = FROM

	is_triggered_only = yes

	option = {
		name = FASCINATING
		hidden_effect = {
			from = {
				clear_deposits = yes
			}
		}
		from = {
			set_deposit = d_living_metal_deposit
		}
		owner = {
			if = {
				limit = {
					has_tech_option = tech_mine_living_metal
				}
				add_tech_progress = {
					tech = tech_mine_living_metal
					progress = 0.25
				}
			}
			else = {
				add_research_option = tech_mine_living_metal
			}
		}
	}
}
