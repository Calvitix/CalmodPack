	is_triggered_only = yes

		start_situation = {
			type = xvcv_mdlc_situation_machine_learning
			target = capital_scope

			set_situation_approach = xvcv_mdlc_situation_machine_learning_approach
		}
	}
}

	is_triggered_only = yes

	}
}
