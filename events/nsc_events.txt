#the event that spawns Flagships for a random set of AI empires.
country_event = {
	id = nsc_events.15
	title = "nsc_events.15.name"
	desc = "nsc_events.15.desc"
	picture = GFX_evt_unknown_ships
	show_sound = event_bad_omen
	is_triggered_only = yes

	immediate = {
		every_rival_country = {
			limit = {
				is_ai = yes
				OR = {
					is_country_type = default
					is_country_type = fallen_empire
					is_country_type = awakened_fallen_empire
				}
				NOT = { any_owned_ship = { is_ship_size = Flagship } }
			}
			nsc_flagship_creation_for_ai_effect = yes
		}
		every_neighbor_country = {
			limit = {
				is_ai = yes
				OR = {
					is_country_type = default
					is_country_type = fallen_empire
					is_country_type = awakened_fallen_empire
				}
				NOT = { any_owned_ship = { is_ship_size = Flagship } }
			}
			nsc_flagship_creation_for_ai_effect = yes
		}
		while = {
			random_country = {
				limit = {
					is_ai = yes
					OR = {
						opinion_level = { who = root level = poor }
						opinion_level = { who = root level = terrible }
					}
					NOT = { any_owned_ship = { is_ship_size = Flagship } }
				}
				nsc_flagship_creation_for_ai_effect = yes
			}
			count = 3
		}
	}

	option = {
		name = nsc_events.15.a
	}
}

			name = NSC_FLAGSHIP_REBUILD_PROJECT

		}
	}
	option = {
		name = nsc_events.16.b #no
		add_resource = { influence = 200 }
	}

		any_owned_ship = { is_ship_size = Flagship }

		clone_leader = {
			target = ruler
			class = admiral
			traits = {
				trait = leader_trait_flagship
			}
			effect = {
				prev.ruler = { set_leader_flag = nsc_flagship_ruler }
				set_leader_flag = nsc_flagship_ruler_leader
				save_event_target_as = nsc_flagship_ruler_leader_target
				if = {
					limit = { has_skill < 3 }
					set_skill = 3
				}
			}
		}
		random_owned_fleet = {
			limit = { any_owned_ship = { is_ship_size = Flagship } }
			assign_leader = event_target:nsc_flagship_ruler_leader_target
			settings = { can_change_leader = no }
		}

	}
