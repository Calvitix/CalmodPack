country_event = {
	id = fix_ai_actions.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		if = {
			limit = {
				NOT = {
					has_global_flag = ai_and_player_flag
				}
			}
			is_ai = yes
		}
		has_global_flag = ai_unemployment_flag
	}

	immediate = {
		random_owned_planet = {
			limit = {
				num_unemployed > 0
			}
			random_owned_pop = {
				limit = {
					is_unemployed = yes
					NOT = {
						has_job = bio_trophy
					}
				}
				prevprev = {
					random_owned_planet = {
						limit = {
							is_colony = yes
							free_jobs > 0
						}
						resettle_pop = {
							pop = prevprev
							planet = this
						}
					}
				}
			}
		}
	}
}

					}

			}
		}

			limit = {
