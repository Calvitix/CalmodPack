volatile_motes_expenditure_starbases = {
	resource = volatile_motes
	type = expenditure
	category = starbases

	potential = {
		any_owned_nonprimary_starbase = {
			has_starbase_size >= orbital_ring_tier_2
			exists = planet
		}
	}

	weight = {
		weight = 0.2

		modifier = {
			factor = 0
			NOT = {
				any_owned_nonprimary_starbase = {
					exists = planet
					OR = {
						AND = {
							has_starbase_size = orbital_ring_tier_2
							count_starbase_buildings = {
								count = 0
							}
						}
						AND = {
							has_starbase_size = orbital_ring_tier_3
							count_starbase_buildings = {
								count < 2
							}
						}
					}
				}
			}
		}
	}
}
