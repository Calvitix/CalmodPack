mt_is_hostile_neighbor = 
{	
	is_country_type = default
	has_communications = PREV	
	is_neighbor_of = PREV				
	OR = 
	{
		is_rival = PREV
		is_hostile_to = prev
		is_unfriendly_to = prev
		AND = 
		{
			is_ai = no
			NOR = 
			{
				has_non_aggression_pact = PREV
				has_defensive_pact = PREV
				is_in_federation_with = PREV
				has_association_status = PREV
				is_subject = yes
			}

		}
	}
}
