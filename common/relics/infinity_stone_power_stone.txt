# Power Stone ##################################################
r_power_stone = {
	activation_duration = 3600
	portrait = "GFX_relic_power_stone"
	sound = "relic_activation_power_stone"

	resources = {
		category = relics
		# Activation cost
		cost = {
			influence = 100
		}
	}

	ai_weight = {
		weight = 10
		modifier = {
			factor = 200
			is_at_war = yes
		}
	}
	triggered_country_modifier = {
		potential = {
			always = yes
		}
		country_energy_produces_mult = 0.20
	}

	score = 5000

	active_effect = {
		add_modifier = {
			modifier = "power_stone"
			days = 3600
		}
		#custom_tooltip = power_stone_activated
		custom_tooltip = relic_triumph_cooldown
		hidden_effect = {
			add_modifier = {
				modifier = "infinity_stone_relic_activation_cooldown" 
				days = 3600
			}
			add_modifier = {
				modifier = "relic_activation_cooldown"
				days = 1
			}
			country_event = { id = infinity_stones_new_gauntlet.1 days = 3600 }
			infinity_circuit_apply_modifier = yes
		}
	}

	# Possible check for activation
	possible = {
		custom_tooltip = {
			fail_text = "requires_infinty_stone_no_cooldown"
			NOT = { has_modifier = infinity_stone_relic_activation_cooldown }
		}
		custom_tooltip = {
			fail_text = "requires_relic_power_stone"
			NOT = { has_modifier = power_stone }
		}
	}
}
