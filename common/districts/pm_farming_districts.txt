	base_buildtime = 240
	icon = district_farming
	is_capped_by_modifier = no

		}
	}

		OR = {

	}
	
	allow = {

	}

	}
	
	resources = {

		}
		upkeep = {
			energy = 1
		}
	}
	
	planet_modifier = {
		planet_housing_add = 10
	}
		
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_gestalt = no
			}
		}
		modifier = {
			job_farming_farmer_alternative_add = 5
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_gestalt = yes
			}
		}
		modifier = {
			job_agri_drone_add = 5
		}
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_gestalt = no
			}
		}
		text = job_farmer_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_gestalt = yes
			}
		}
		text = job_agri_drone_effect_desc
	}
	
	prerequisites = {
	}
	
	ai_weight = {
		weight = 1
		modifier = {
			factor = 3
			exists = owner #Calvitix
			owner = { 
				has_monthly_income = {
					resource = food
					value < 0
				}
			}
		}

		modifier = {
			factor = 3
			free_housing < 0
		}

		modifier = {
			factor = 3
			any_owned_pop = { 
				OR = { 
					is_unemployed = yes 
					has_job = servant 
				}
			}
		}

		# No one to work it anyway
		modifier = {
			factor = 0
			free_housing > 0
			NOT = {
				any_owned_pop = { 
					OR = { 
						is_unemployed = yes 
						has_job = servant 
					}
				}
			}
		}
	}

	ai_resource_production = {
		food = 1
		trigger = {
			always = yes
		}
	}
}
