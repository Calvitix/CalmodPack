# Sets the edict-cost for all edicts with this variable
@campaignCost = 1500
@campaignDuration = 3600

@suppressCampaignCost = 2000
@suppressCampaignDuration = 3600
@researchTimedFlag = 7200

#################
# Campaigns
# shorter duration, cost energy
#################

eac_physics_grant = {
	length = @campaignDuration
	prerequisites = { eac_tech_boost_research }
	resources = {
		category = campaigns 
		cost = { energy = @campaignCost }
	}

	modifier = {
		country_physics_tech_research_speed = 0.15
	}
	
	allow = {
		custom_tooltip = {
			fail_text = eac_grant_already_done
			OR = {
				NOR = {
						has_edict = eac_society_grant
						has_edict = eac_engineering_grant
				}
				has_edict = eac_physics_grant
			}
		}		
	}

	potential = {
		NOT = { has_ethic = ethic_gestalt_consciousness }
	}
	
	ai_weight = {
		weight = 0
	}	
}

eac_suppress_xenophobe = {
	length = @suppressCampaignDuration
	prerequisites = { eac_tech_suppress_ethics }
	resources = {
		category = campaigns 
		cost = { 
			energy = @suppressCampaignCost
			influence = 10 
		}
	}

	modifier = {
		pop_ethic_xenophobe_attraction_mult = -0.5
	}

	show_tech_unlock_if = {
		NOR = {
			has_ethic = ethic_xenophobe
			has_ethic = ethic_fanatic_xenophobe
		}
		always = no
	}

	effect = {
		hidden_effect = {
			every_owned_pop = {
				limit = { 
					OR = {
						has_ethic = ethic_xenophobe
						has_ethic = ethic_fanatic_xenophobe				
					}
				}
				eac_ace_set_ethic_pop_unhappyness = yes
			}
		}
	}
	
	potential = {
		eac_has_suppression_authority = yes
		NOR = {
			has_ethic = ethic_xenophobe
			has_ethic = ethic_fanatic_xenophobe
		}
		is_ai = no
	}
	
	ai_weight = {
		weight = 0
	}
		
}
