@suppressCampaignDuration_base = 3600
@suppressCampaignDuration_civic = 4500
@suppressCampaignDuration_perk = 5400
@suppressCampaignDuration_both = 6300

eac_ace_set_ethic_pop_unhappyness = {
    owner = {
        if = { 
            limit = {
                NOR = {
                    has_civic = civic_imperial_cult
                    has_ascension_perk = ap_executive_vigor
                }
            }
            PREV = { 
                add_modifier = { modifier = eac_suppressed_ethic days = @suppressCampaignDuration_base } 
            }
        }
        else_if = { 
            limit = {
                has_civic = civic_imperial_cult
                NOT = { has_ascension_perk = ap_executive_vigor }
            }
            PREV = { 
                add_modifier = { modifier = eac_suppressed_ethic days = @suppressCampaignDuration_civic }
            }
        }
        else_if = { 
            limit = {
                has_ascension_perk = ap_executive_vigor
                NOT = { has_civic = civic_imperial_cult }
            }
            PREV = { 
                add_modifier = { modifier = eac_suppressed_ethic days = @suppressCampaignDuration_perk }
            }
        }
        else_if = { 
            limit = {
                has_ascension_perk = ap_executive_vigor
                has_civic = civic_imperial_cult
            }
            PREV = { 
                add_modifier = { modifier = eac_suppressed_ethic days = @suppressCampaignDuration_both }
            }
        }
    }
}
