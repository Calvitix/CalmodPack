
eco_check_tiyanki_population = {
	if = {
		limit = { is_country_type = tiyanki }
		if = {
			limit = { check_variable = { which = eco_population value > 5 } }
			
			if = { 
				limit = { has_global_flag = eco_tiyanki_endangered }
				remove_global_flag = eco_tiyanki_endangered
			}
		} else_if = {
			limit = { check_variable = { which = eco_population value > 0 } }
			
			if = { 
				limit = { has_global_flag = eco_tiyanki_extinct }
				remove_global_flag = eco_tiyanki_extinct
			}
		}
		if = {
			limit = { check_variable = { which = eco_population value < 1 } }
			
			if = { 
				limit = { NOT = { has_global_flag = eco_tiyanki_extinct } }
				set_global_flag = eco_tiyanki_extinct
			}
		} else_if = {
			limit = { check_variable = { which = eco_population value < 6 } }
			
			if = { 
				limit = { NOT = { has_global_flag = eco_tiyanki_endangered } }
				set_global_flag = eco_tiyanki_endangered
			}
		}
	}
}
